73  int i;

75  ssSetNumSFcnParams(S, N_S_FUNCTION_PARMS); /* Number of exp. parameters */
76  if (ssGetNumSFcnParams(S) != ssGetSFcnParamsCount(S)) {
77  ssSetErrorStatus(S,"Incorrect number of s-function parameters");
78  return;
79  }
80  ssSetSFcnParamTunable(S, 0, 0); /* all parameters are constant */

88  for lookup
89  port 0: state vector for lookup
90  port 1: command vector for lookup
91  port 2: state vector for learning
92  port 3: command vector for learning
93  port 4: target vector for learning

96  if (!ssSetNumInputPorts(S, N_S_FUNCTION_INPUT_PORTS))

102  ssSetInputPortWidth(S, COMMAND_LEARN_PORT, n_commands);
103  ssSetInputPortWidth(S, TARGET_LEARN_PORT, n_targets);
104  ssSetInputPortWidth(S, TRACKING_ERROR_PORT, n_targets);
105 
106  for (i=0; i<N_S_FUNCTION_INPUT_PORTS; ++i) {
107  ssSetInputPortRequiredContiguous(S, i, true);
108  ssSetInputPortDirectFeedThrough(S, i, false);
109  }
110  ssSetInputPortDirectFeedThrough(S, STATE_PREDICT_PORT, true);
111  ssSetInputPortDirectFeedThrough(S, COMMAND_PREDICT_PORT, true);

119  return;

225  ssSetErrorStatus(S,errormsg);
226  return;
227  }

255  int n_states = (int) n_states_ptr[0];
256  int n_commands = (int) n_commands_ptr[0];

259  int lwpr_ID = (int) lwpr_ID_ptr[0];

262  const real_T *u = (const real_T*) ssGetInputPortSignal(S,COMMAND_PREDICT_PORT);
263  real_T *y = ssGetOutputPortSignal(S,FUNCTION_PREDICT_PORT);

265  real_T *t = ssGetOutputPortSignal(S,TARGET_PORT);
266 
267  /* local variables */
268  static double cutoff = 0.001;
269  static Matrix out=NULL;
270  static iMatrix ypart=NULL;
271  static int n_in_allocated=0;
272  static int n_out_allocated=0;
273  static int n_predictions_allocated=0;
274 
275  int n_in;
276  int n_out;
277  int flag=FALSE;
278  int delta_predictions=0;
279  int i,j,count;
280  int linear_gx;
281 
282  /* given that we share the in, out, and ypart variables across all LWPR
283  models in one simulink simulation, we need to ensure that they are
284  sufficiently dimensioned */
285  n_in = lwprs[lwpr_ID].n_in_reg;
286  n_out = lwprs[lwpr_ID].n_out;
287  linear_gx = lwprs[lwpr_ID].linear_gx;
288 
289  if (n_out > n_out_allocated || n_predictions > n_predictions_allocated) {
290  if (out==NULL)
291  out = my_matrix(1,n_out,1,n_predictions);
292  else
293  mat_add_shape(&out, n_out-n_out_allocated,n_predictions-n_predictions_allocated);
294  delta_predictions = n_predictions-n_predictions_allocated;
295  n_out_allocated = n_out;
296  n_predictions_allocated = n_predictions;
297  }
298 
299  if (delta_predictions!=0 || n_in > n_in_allocated) {
300  if (ypart==NULL)
301  ypart = my_imatrix(1,n_predictions,1,n_in+1);
302  else
303  imat_add_shape(&ypart, delta_predictions,n_in-n_in_allocated);
304  n_in_allocated = n_in;
305  }
306 
307  if (n_part_mat != n_predictions*(n_in+1)) {
308  sprintf(errormsg,
309  "Number of elements in partion matrix (=%d) does not match n_precitions*(n_in+1) (=%d)\n",
310  n_part_mat,n_predictions*(n_in+1));
311  ssSetErrorStatus(S,errormsg);
312  return;
313  }
314 
315  /* assign the partition matrix */
316  count=0;
317  for (i=1; i<=n_in+1; ++i)
318  for (j=1; j<=n_predictions; ++j)
319  ypart[j][i] = part_mat[count++];
320 
321  predictLWPROutputPartCompControl(lwpr_ID,(double *)(x-1),(double *)(u-1),cutoff,TRUE,
322  n_predictions,ypart,out,&(rfID_predict[lwpr_ID]));
323 
324  count = 0;
325  for (i=1; i<=n_out; ++i)
326  for (j=1; j<=n_predictions; ++j)
327  y[count++] = out[i][j];
328 
329  /* the predicted outputs and training outputs are kept as internal state, such
330  that they can serve as output in mdlOutput. Keeping as the the training
331  data as internal state is to avoid a time mal-alignment between prediction
332  and training output */
333 
334  for (i=1; i<=n_out; ++i) {
335  tp[i-1] = disc_states[i-1];
336  t[i-1] = disc_states[n_out+i-1];
337  }
338 
339 }
